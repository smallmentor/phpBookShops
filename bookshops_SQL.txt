-- ===========================================
-- |               Writers 作者              |
-- ===========================================

create table Writers
(W_ID VARCHAR(10) NOT NULL,
 W_NAME VARCHAR(50) NULL,
 PRIMARY KEY(W_ID)
);

-- ===========================================
-- |         Publishing_House 出版社         |
-- ===========================================

create table Publishing_House 
(PH_ID VARCHAR(8) NOT NULL,
 PH_NAME VARCHAR(50) NULL,
 PH_PHONE VARCHAR(15) NULL,
 PH_ADDRESS VARCHAR(50) NULL,
 PRIMARY KEY(PH_ID)
);

-- ===========================================
-- |                 Books 書                |
-- ===========================================

create table Books
(ISBN VARCHAR(15) NOT NULL,
 B_NAME VARCHAR(50) NULL,
 B_DATE DATE NULL,
 B_PRICE INT(10) NULL,
 B_CLASS VARCHAR(10) NULL,
 W_ID VARCHAR(10) NULL,
 PH_ID VARCHAR(8) NULL,
 PRIMARY KEY(ISBN),
 FOREIGN KEY(W_ID) REFERENCES Writers(W_ID),
 FOREIGN KEY(PH_ID) REFERENCES Publishing_House(PH_ID)
);
